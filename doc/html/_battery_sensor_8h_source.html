68 class InternalVCC {
69 public:
70  typedef uint16_t ValueType;
71  static const int DefaultDelay = 0;
72 
73  void init () {
74 #ifdef ARDUINO_ARCH_STM32F1
75  adc_reg_map *regs = ADC1->regs;
76  regs->CR2 |= ADC_CR2_TSVREFE; // enable VREFINT and temp sensor
77  regs->SMPR1 = ADC_SMPR1_SMP17; // sample rate for VREFINT ADC channel
78 #endif
79  }
80 
81  void start () {}
82 
83  uint16_t finish () {
84  uint16_t vcc=0;
85 #ifdef ARDUINO_ARCH_AVR
86  // Read 1.1V reference against AVcc
87  // set the reference to Vcc and the measurement to the internal 1.1V reference
88  ADMUX &= ~(ADMUX_REFMASK | ADMUX_ADCMASK);
89  ADMUX |= ADMUX_REF_AVCC; // select AVCC as reference
90  ADMUX |= ADMUX_ADC_VBG; // measure bandgap reference voltage
91  _delay_us(350);
92  ADCSRA |= (1 << ADSC); // start conversion
93  while (ADCSRA & (1 << ADSC)) ; // wait to finish
94  vcc = 1100UL * 1024 / ADC;
95 #elif defined ARDUINO_ARCH_STM32F1
96  vcc = 1200 * 4096 / adc_read(ADC1, 17); // ADC sample to millivolts
97 #endif
98  DPRINT(F("iVcc: ")); DDECLN(vcc);
99  return vcc;
100  }
101 };

128  InternalVCC::start(); // in case we add something here later

130  }

358  //DDECLN(__gb_BatCurrent);
359  ATOMIC_BLOCK( ATOMIC_RESTORESTATE ) {
360  __gb_BatIgnore = 10; // first 10 values will be ignored
361  __gb_BatIrq = irq;
362  }
363  ADMUX &= ~(ADMUX_REFMASK | ADMUX_ADCMASK);
364  ADMUX |= ADMUX_REF_AVCC; // select AVCC as reference
365  ADMUX |= ADMUX_ADC_VBG; // measure bandgap reference voltage
366  ADCSRA |= (1 << ADIE) | (1<<ADPS0) | (1<<ADPS1) | (1<<ADPS2); // enable interrupt & 128 prescaler
367  ADCSRA |= (1 << ADSC); // start conversion
368  }
371  static void irq () {
372  __gb_BatCount++;
373  if( __gb_BatIgnore > 0 ) {
374  __gb_BatIgnore--;
375  }
376  else {
377  uint16_t v = 1100UL * 1024 / ADC;
378  if( __gb_BatCurrent == 0 ) {
379  __gb_BatCurrent = v;
380  }
381  else {
382  v = (__gb_BatCurrent + v) / 2;
383  if( v < __gb_BatCurrent ) {
384  __gb_BatCurrent = v;
385  }
386  }
387  }

390  ADCSRA |= (1 << ADSC); // start conversion again
